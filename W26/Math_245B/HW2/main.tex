\section{Exercise 2.1}
\emph{
    Let $\mu $ be a Borel (outer) measure on a separable metric space $X$.
    Let $V$ be the union of all open sets $O \subset X$ such that $\mu(O) = 0$
    (spt $\mu  := X \setminus V$ is called the support of $\mu $ ).
    \begin{enumerate}
        \item Show that $X$ is second countable (has a countable basis of open sets)
        \item Show that $\mu(X \setminus \spt \mu) = 0$
        \item Further assume that $\mu(X) < \infty$. Show that $\spt \mu$ is the smallest closed subset $K$ such that
             $\mu(K) = \mu(X)$.
    \end{enumerate}
}
\solution{
    \begin{enumerate}
        \item
            Since $X$ is separable there exists a countable dense subset $Q \subset X$ such that for all $x \in X, \epsilon > 0, \exists q \in Q, d(q,x) < \epsilon$.
            Let $Q = \{q_i\}_{i=1}^{\infty}$ and Define
            \[
                B := \{D(q_i,\frac{1}{m}) : i,m \in \mathbb{N}\}
            \] 
            where $D(x,r)$ is the open ball in  $X$ centered at  $x$ of radius $r$. I claim that $B$ is a base of $X$.
            Indeed let  $O \subset X$ be open then for each $x \in O$ there exists  $\epsilon(x) >0$ such that $D(x,\epsilon(x)) \subset O$,
            and there exists $q(x) \in Q$ such that  $d(x,q(x)) < \frac{\epsilon(x)}{3}$ and $m(x)$ such that  $\frac{\epsilon(x)}{3} \le \frac{1}{m(x)} < \frac{\epsilon}{2}$ together we get that 
            \[
                O = \bigcup_{x \in O}D(q(x),\frac{1}{m(x)})
            \] 
            since each $x \in O$ is in the ball  $D(q(x),\frac{1}{m(x)})$ which itself is contained in $D(x,\epsilon(x)) \subset O$. Also $B$ is a countable collection of sets since  $Q$ is countable, so $B$ is a countable base. 
        \item 
            we have
            \[
            X \setminus \spt \mu = X \setminus (X \setminus V) = V
            \] 
            \[
                V = \bigcup_{\substack{O \subset X \\ \mu(O) = 0}}O = \bigcup_{\substack{O \subset X\\ \mu(O) = 0}}\bigcup_{\substack{B_i \in B\\ B_i \subset O}}B_i = \bigcup_{j \in M \subset \mathbb{N}}^{}B_j
            \] 
            for some collection $M$ of elements of the basis from the last question $B$. Then we have by countable subadditivity of measure
            \[
                \mu(V) = \mu(\bigcup_{j \in M}B_j) \le \sum_{j \in M}^{}\mu(B_j) = 0
            \] 
            since each $B_j \subset O$ so $\mu(B_j) \le \mu(O) = 0$.
        \item Assume $\mu(X) < \infty$ then we have
            \[
            \mu(\spt \mu) = \mu(X \setminus V) = \mu(X) - \mu(V) = \mu(X)
            \] 
            Assume for the sake of contradiction that there exists $K \subset \spt \mu$ $K \ne \spt \mu$ obeying $\mu(X) = \mu(K)$
            then
            \[
            \mu(X \setminus K) = \mu(X) - \mu(K) = 0
            \] 
            so $K^{c}$ is an open set with measure 0 and $V \subset K^{c}$, however $K^{c}$ is open and so $K^{c} \subset V$ thus $K^{c} = V$ and $K = V^{c} = \spt \mu$ thus $\spt \mu$ is the smallest closed subset that has full measure.
    \end{enumerate}
}

\section{Exercise 2.4}
\emph{
    Let $\mathcal{L}$ be the class of Lebesgue measurable sets in  $ \mathbb{R}^{d}$ and let $\mu$
    be the Lebesgue measure on $ \mathbb{R}^{d}$. Let $(\nu_j)_j$ be a monotone sequence of Radon measures on  $ \mathbb{R}^{d}$ such
    that $\nu(E) := \lim_{j\to \infty}\nu_j(E)$ is a Radon measure on $ \mathbb{R}^{d}$. Show that
    \[
    D_\mu\nu = \lim_{j\to \infty} D_\mu\nu_j \quad a.e.
    \] 
}
\solution{
    Assume first that the sequence $(\nu_j)_j$ is monotone increasing, that is for all  $A \subset \mathbb{R}^{d}$ $\nu_j[A] \le \nu_{j+1}[A]$ then we have similarly $\nu_j[A] \le \nu[A]$ and so for all $r$
\[
    \frac{\nu(B_r(x))}{\mu(B_r(x))} \ge \frac{\nu_j(B_r(x))}{\mu(B_r(x))} \implies D_\mu \nu \ge D_\mu \nu_j
\] 
taking limits we get
\[
D_\mu \nu \ge \lim_{j\to \infty} D_\mu \nu_J
\] 
We know via the lebesgue decomposition theorem that there exists a set $B \subset \mathbb{R}^{d}$ such that $\mu[B^{c}] = 0$ and
\[
    \nu = \nu|_B + \nu|_{B^{c}} = \nu_{abs} + \nu_s \qquad \nu_{abs} << \mu \qquad \nu_{s} \perp \mu
\] 
There also exists a sequence of sets $(B_j)_j \subset 2^{\mathbb{R}^{d}}$ such that $\mu[B_j^{c}] = 0$ for all $j$
\[
    \nu_j = \nu_j|_{B_j} + \nu_j|_{B_J^{c}}
\] 
with the absolutely continuity and singularity holding as well. Then $\mu[(B \cap \bigcap_{j=1}^{\infty}B_j)^{c}] = 0$
so it suffices to show the inequality on $\tilde B = B \cap \bigcap_{j=1}^{\infty}B_j$ since if it holds almost everywhere on  $\tilde B$ then it holds $\mu$-almost everywhere. Let $E \subset \tilde B$ be a $\mu$-measurable set then
\[
    \int_{E}^{} D_\mu \nu(x) d \mu = \nu[E] = \lim_{j\to\infty }\nu_j[E] = \lim_{j\to\infty } \int_{E}^{}D_\mu \nu_j(x) d \mu = \int_{E}^{} \lim_{j\to \infty} D_\mu \nu_j(x) d \mu
\] 
Firstly by lebesgue differentiation theorem since when we restrict $\nu_j$ and  $\nu$ to  $\tilde B$ they are absolutely continuous. Further the other inequality is ipmlied by the monotone convergence theorem since $D_\mu \nu_j$ is a monotone increasing sequence of functions with respect to $j$
\[
\int_{E}^{}D_\mu \nu d \mu = \int_{E}^{} D_\mu \nu_j d \mu
\] 
for all $E \subset \tilde B$ which implies that
\[
D_\mu \nu =  \lim_{j\to \infty} D_\mu \nu_j 
\] 
$\mu$-almost everywhere on $\tilde B$ which further implies that  $D_\mu \nu = D_\mu \nu_j$ $\mu$-almost everywhere since $\tilde B^{c}$ is a null set.
For the other possiblility, the $(\nu_j)_j$ are monotone decreasing, keeping $\tilde B$ defined as before, we further restrict ourselves to proving that for all  $E \subset \tilde B$
 \[
\int_{E \cap B_n(0)}^{}D_\mu \nu(x)d \mu = \int_{E \cap B_n(0)}^{} \lim_{j\to \infty} D_\mu \nu_j(x) d \mu
\] 
after which, we conclude that the two functions agree on all bounded sets and so the set of which they agree almost sure is the union of $E \cap B_n(0)$ over all  $n \in \mathbb{N}$ and we ultimately conclude that they agree almost everywhere on all of $ \mathbb{R}^{d}$.
To prove this equality we cite the dominated convergence theorem to state that for fixed $E$
 \[
\int_{E \cap B_n(0)}^{}D_\mu \nu_j(x) d \mu \le \int_{ E \cap B_n(0)}^{} D_\mu \nu_1(x) d \mu < \infty
\] 
With the inequality holding pointwise as well.
since $\nu$ is a Radon measure so the integrals are finite on all compact sets. Thus by dominated convergence we argue
\[
    \int_{E \cap B_n(0)}^{} D_\mu \nu(x) d \mu = \lim_{j\to \infty}\int_{E \cap B_n(0)}^{}D_\mu \nu_j(x) d \mu = \int_{E \cap B_n(0)}^{}\lim_{j\to \infty}D_\mu \nu_j d \mu
\] 
So we conclude as before that the functions are the same almost everywhere.
}

\section{Exercise 2.6}
\emph{
    Let $a < b$ be real numbers and let $F : [a,b] \rightarrow \mathbb{R}$ be a continuous function
    \begin{enumerate}
        \item Show that the set $A$ of points of differentiablility of $F$ is a Borel set.
        \item Suppose that $F$ is monotone nondecreasing and $\mu_F$ is the associated Lebesgue-Stieltjes measure.
            Let $B_{\infty} := \{F' = +\infty\}$. Show that for any Borel set  $E$ 
            \[
                \mu_F(E) = \int_{E}^{}F'(t)dt + \mu_F(E \cap B_{\infty})
            \] 
    \end{enumerate}
    Similar results can be obtained when $F$ is of bounded variations and $\mu_F$ is a signed measure
}
\solution{
    I define two functions
    \[
        \overline{D}(x) : (a,b) \rightarrow \overline{\mathbb{R}} \qquad \overline{D}(x) = \limsup_{h\to 0}\frac{F(x+h) - F(x)}{h}
    \] 
    and
    \[
        \underline{D}(x): (a,b) \rightarrow \overline{ \mathbb{R}} \qquad \underline{D}(x) = \liminf_{h\to 0}\frac{F(x+h) - F(x)}{h}
    \] 
    then, the set of points where the two functions agree and are not infinite is the set of continuity points. First I show that
    $\overline{D}(x)$ is a Borel function. To do this we verify that the preimage of level sets is Borel.
    \[
        \{x : \overline{D}(x) \ge L\} = \{x : \inf_{\delta > 0} \sup_{0 < |h| < \delta} \frac{F(x+h) - F(x)}{h} \ge L\}
    \] 
    \[
        = \{x : \forall \delta > 0, \exists 0 < |h| < \delta \frac{F(x+h) - F(x)}{h} \ge L\} = \bigcap_{m=1}^{\infty}\bigcup_{\substack{q\in \mathbb{Q} \\ 0 < |q| < \frac{1}{m}}}^{} \bigcap_{k=1}^{\infty} \{x : \frac{f(x+q)-f(x)}{q} \ge L - \frac{1}{k}\}
    \] 
    where we use continuity of $F$ to reduce checking over all  $\delta$ and $h$ to only checking over countably many combinations. This works since the supremums are monotonically decreasing as $\delta$ shrinks so taking a countable subsequence of $\frac{1}{m}$ going to zero achieves the same limit.
    Additionally since $F$ is continuous we for any point achieving the supremum and any  $\epsilon$ there exists a rational point that is within that epsilon, that is why we added the third term.
    Since the function 
    \[
    \frac{f(x+h)-f(x)}{h}
    \] 
    is continuous and therefore borel measurable for any fixed $h > 0$ we conclude that the level sets of $\overline{D}(x)$ are Borel. Similarly, the level sets of $\underline{D}(x)$ are Borel. We then construct the set of differentiable points
     \[
        \{x: F'(x) \text{ exists }\} = \{x : \overline{D}(x) = \underline{D}(x)\} \setminus \{x : \overline{D}(x) = +\infty\}
    \] 
    after disregarding all points where the up limit is infinite we can take the terms to the other side and conclude that the first set is a level set of $\overline{D}(x) - \underline{D}(x)$ so it is a Borel set. We then conclude that the set of differentiability points is Borel.
    Now set
    \[
    F'(x) =
    \limsup_{h\to 0} \frac{f(x+h)-f(x)}{h} = \limsup_{r\to 0} \frac{\mu_F(B_r(x))}{\mathcal{L}(B_r(x))}
    \] 
    then $\mu_F = \mu_F^{abs} + \mu_F^{s}$ where $\mu_F^{abs} << \mathcal{L}$ and $\mu_F^{s} \perp \mathcal{L}$ and by the Lebesgue decomposition theorem we have
    \[
        \mu_F(E) = \int_{E}^{}D_{\mathcal{L}}\mu_F(x)dx + \mu_F^{s}(E)
    \] 
    Now if I can show that $\mu_F^{abs} = \mu_F|_{\{F' \ne +\infty\}}$ then the theorem is complete since $F' = D_{\mathcal{L}}\mu_F$ where $F' \ne +\infty$.
    First I show that
    \[
        \mathcal{L}\{x : \overline{D}(x) = +\infty\} = 0
    \] 
    indeed for any fixed $n$
    \[
        A = \{X : \overline{D}(x) = +\infty\} \cap [-r,r]
    \] 
    then by Lemma proved in class
    \[
        \mu_F[[-r,r]] = F(r) - F(-r) \ge \mu_F[A] \ge \alpha \mu[A]
    \] 
    for all $\alpha$ and so $\mu[A] = 0$ and since this holds for all $r$ by continuity from below of measure we have  $\mu[\{x: \overline{D}(x) = +\infty\}] = 0$.
    Fix $E$ a borel set in $ \mathbb{R}$ then by letting $B_\infty := \{F' = \infty\}$
    \[
    \mu_F(E) = \mu_F(E \setminus B_\infty) + \mu_F(E \cap B_\infty)
    \] 
    Consider for any $t \in (1,\infty)$
    \[
        \mu_F(E \setminus B_\infty) = \bigcup_{n \in \mathbb{Z}}^{\infty} \mu_F(E \cap \{F' \in [t^{n},t^{n+1})\}) + \mu_F(E \cap \{F' = 0\})
    \] 
    On a fixed $n$ we have that
    \[
        t^{n} \mathcal{L}(E \cap \{F' \in [t^{n}, t^{n+1})\}) \le \mu_F(E \cap \{F' \in [t^{n}, t^{n+1})\} \le t^{n+1} \mathcal{L}(E \cap \{F' \in [t^{n},t^{n+1})\})
    \] 
    so clearly $\mu_F|_{0 < F' < \infty} << \mathcal{L}$. It suffices finally to prove that $\mu_F\{x : F' = 0\} = 0$ indeed
    \[
        \mu_F[B_r(0) \cap \{F' = 0\}] \le \alpha \mathcal{L}[B_r(0) \cap \{F' = 0\}] < \infty
    \] 
    for all $\alpha > 0$ and so by continuity from below we have
    \[
        \mu_F[\{F' = 0\}] = 0
    \] 
    thus we have
    \[
        \mu[E] = \mu[E \cap \{F' < \infty\}] + \mu[E \cap \{F' = \infty\}] = \int_{E}^{} F'(x) dx + \mu_F[E \cap \{F' = \infty\}]
    \] 
}

\section{Exercise 2.7}
\emph{
    Let $(\nu_j)_j$ be a sequence of Borel measures on a metric space  $\Omega$ and let  $\mu$ be a Borel measure on $\Omega$. Show that if
     $\nu_j \perp \mu$ for all $j$ then $\sum_{j=1}^{\infty}\nu_j \perp \mu$.
}
\solution{
    Since each $\nu_j \perp \mu$ there eixsts a sequence $(B_j)_j$ such that  $\mu(B_j) = 0$ and $\nu_j(B_j^{c}) = 0$ Then $\mu(\bigcup_{j=1}^{\infty}B_j) = 0$
    and $\nu_j((\bigcup_{i=1}^{\infty}B_i)^{c}) = \nu_j(\bigcap_{i=1}^{\infty}B_i^{c}) = 0$ so we conclude that
    \[
        \sum_{j=1}^{\infty}\nu_j(\bigcap_{i=1}^{\infty}B_i^{c}) = \lim_{n\to \infty}\sum_{j=1}^{n}\nu_j(\bigcap_{i=1}^{\infty}B_i^{c}) = \lim_{n\to \infty}\sum_{j=1}^{n}0 = 0
    \] 
    the desired conclusion.
}

\section{Exercise 2.8}
\emph{
    Let $X = [0,1], \Sigma$  is the  $\sigma$ algebra of Borel subsets of $X, m := \mathcal{L}^{1}|_X$ is the Lebesgue measure on $X$ and  $\mu$
    is the counting measure.
    \begin{enumerate}
        \item Show that $m << \mu$ but there is no $f$ such that $f = \frac{d m}{d \mu}$.
        \item Show that $\mu$ has no Lebesgue decomposition with respect to $m$
    \end{enumerate}
}
\solution{
    \begin{enumerate}
        \item To show that $m << \mu$ Let $A \subset \mathbb{R}^{d}$ be such that $\mu[A] = 0$ then $A = \emptyset$ so we have  $m([A]) = 0$.
            Assume for contradiction that there existed  $f = \frac{d m}{d \mu}$ then
            \[
                0 =  m[\{x\}] \int_{\{x\}}^{}f d \mu = f
            \] 
            so $f = 0$ on  $[0,1]$ but this contradicts
             \[
                 \frac{1}{2} = m[(0,\frac{1}{2})] = \int_{(0,\frac{1}{2})}^{}f d \mu = \int_{(0,\frac{1}{2})}^{} 0 d \mu = 0
            \] 
            so no $f$ exists satisfying the radon nikodym derivative.
        \item Assume for contradiction $\mu$ had a Lebesgue decomposition $\mu_{abs} + \mu_s$ with $\mu_{abs} << m$ and $\mu_s \perp m$. Then there exists B such that $\mu_s = \mu|_B$ and there exists further $A$ such that $\mu_s[A] = 0$ and $m[A^{c}] = 0$.
            In particular $A \cap B = \emptyset$ so  $B \subset A^{c}$ and so $m[B] = 0$ and so  $B$ is a null set with respect to lebesgue measure.
            and if it were that
            \[
                \mu[A] = \mu_{acc}[A] + \mu_s[A]
            \] 
            But by letting $A = \{x\}$ for any  $x \in B^{c}$ we see that $\mu[A] = 1$ but $\mu_s[A] = 0$ since $A \cap B = \emptyset$
            and since  $\mu_{acc} << m$ and since $m[A] = 0$ we must have  $\mu_{acc}[A] = 0$ so we get
            \[
            1 = 0 + 0
            \] 
            a contradiction, thus no decomposition exists.
    \end{enumerate}
}

\section{Exercise 2.9}
\emph{
    Let $\mu$ be a Radon measure on $ \mathbb{R}^{d}$ and let $r > 0$. Show that  $x \rightarrow \mu(B(x,r))$ is upper semicontinuous in the sense that $\limsup_{y\to x}\mu(B(y,r)) \le \mu(B(x,r))$
}
\solution{
    if $|y-x| < \delta$ then $B(y,r) \subset B(x,r+\delta)$ this implies that
    \[
    \limsup_{y\to x}\mu(B(y,r)) \le \mu(B(x,r+\delta))
    \] 
    for all $\delta$, letting $\delta \rightarrow 0$ since $\mu$ is Radon we have continuity from above on radius to get
    \[
    \mu(B(x,r+\delta)) \rightarrow \mu(B(x,r))
    \] 
    so
    \[
    \limsup_{y\to x }\mu(B(y,r)) \le \mu(B(x,r+\delta))
    \] 
}

\section{Exercise 2.10}
\emph{
    Let $\mu$ and $\nu$ be  $\sigma$-finite Borel measures on $ \mathbb{R}^{d}$, which are Borel-regular and such that $\nu << \mu$. Show that
    if $\lambda := \mu + \nu$ and $f = \frac{d\nu}{d \lambda}$, then $0 \le f \le 1$  $\mu$ a.e. and $\frac{d \nu}{d  \mu} = \frac{f}{1-f}$
}
\solution{
    By definition
    \[
        \nu[A] = \int_{A}^{}f d \lambda = \int_{A}^{}f(d \nu + d \mu) = \int_{A}^{}f d \nu + \int_{A}^{}f d \mu
    \] 
    If we assume there was a region $A$ such that  $f \in (1,\infty)$ on $A$ we would get a contradiction.
    Since if $A$ has positive  $\nu$ measure then the lhs would be less than rhs and if  $A$ had no  $\nu$ measure but had positive  $\mu$-measure we would 
    get that a null set has positive measure another contradiction so it must be that $f \le 1$  $\mu$-a.e.
    Similarly  $f$ is positive since $\nu$ and  $\mu$ are positive everywhere and so if there was a region
    such that  $f \in (-\infty,0)$ with positive measure under $\mu$ we would similarly reach a contradiction.
    Consider $A = \{x : f(x) = 1\}$, since $\mu$ and $\nu$ are  $\sigma$-finite we can assume that $A$ has finite measure wrt both measures.
    Then we must have
     \[
         \mu[A] = 0
    \] 
    since 
    \[
        \nu[A] = \int_{A}^{}1 d \nu + \int_{A}^{}1 d \mu = \nu[A] + \mu[A]
    \] 
    and so $f \in [0,1)$  $\mu$-almost everywhere.
    Let $g = \frac{d \nu}{d \mu}$ 
    then
    \[
        \nu[A] = \int_{A}^{} g d \mu = \int_{A}^{} f d \nu + \int_{A}^{} f d \mu = \int_{A}^{}fg d \mu + \int_{A}^{}f d \mu
    \] 
    since this holds for all $f$ we get that
     \[
         g = fg + f \qquad \text{ $\mu$-a.e.}
    \] 
    which implies
    \[
        g = \frac{f}{1-f} \qquad \text{$\mu$-a.e.}
    \] 
    thus completing our proof.
    \\
    Alternate solution(not for grading):
    First note that $\nu << \lambda$.
    We have $f = \frac{d \nu}{ d \lambda}$
    \[
        f(x) = 
        \begin{cases}
            \limsup_{r\to 0^{+}} \frac{\nu[B_r(x)]}{\mu[B_r(x)] + \nu[B_r(x)]} & \text{ if $\mu[B_\epsilon(x)] + \nu[B_\epsilon(x)] > 0$ for all $\epsilon > 0$}\\
            +\infty & \text{ there exists $\epsilon_0$ such that $\mu[B_{\epsilon_0}(x)] + \nu[B_{\epsilon_0}(x)] =0$ }
        \end{cases}
    \] 
    where we use the convention that if the numerator and denominator are both infinite we get 1 since the terms in the numerator are always less than or equal to denominator.
    then for fixed $r$ we have 
    \[
        0 \le f \le 1 \qquad \text{ or } f = \infty
    \] 
    We proved in class that the points where the derivative is infinity is a null set with respect to $\mu$ (since null sets with respect to $\lambda$ are null wrt $\mu $ ).
    Additionaly we show that $f < 1$  $\mu$- almost everywhere since if we let $E_i = \{x : f(x) = 1, x \in B_i\}$ where  $\{B_i\}_{i=1}^{\infty}$ is a collection of finite measure sets wrt $\nu$ whose union is all of $ \mathbb{R}^{d}$
    \[
        \nu[E_i] = \int_{E_i}^{}f d\lambda = \int_{E_i}^{}f d \mu + \int_{ E_i}^{} f d \nu = \int_{E_i}^{} d \mu + \int_{E_i}^{} d \nu = \mu[E_i] + \nu[E_i]
    \] 
    subtracting from both sides since $\nu[E_i] < \infty$ we get
    \[
        0 = \mu[E_i]
    \] 
    which implies that $f < 1$  $\mu$-almost everywhere. 
    \[
    g = \frac{d \nu}{d \mu} =
    \begin{cases}
        \limsup_{r\to 0^{+}}\frac{\nu[B_r(x)]}{\mu[B_r(x)]} & \mu[B_\epsilon(x)] > 0 \forall \epsilon > 0\\
        +\infty & \exists \epsilon_0 \rightarrow \mu[B_{\epsilon_0}(x)] = 0
    \end{cases}
    \] 
    in class we showed that $g < \infty$  $\mu$ almost everywhere, so it suffices to show that $g = \frac{f}{1-f}$ when $f$ is less than 1 and when the lim sup = lim inf since we also showed that they agree almost everywhere with respect to $\mu$.
    We have
    \begin{align*}
        \frac{f}{1-f} &= \lim_{r\to 0^{+}} \frac{\frac{\nu[B_r(x)]}{\mu[B_r(x)] + \nu[B_r(x)]}}{1 - \frac{\nu[B_r(x)]}{\mu[B_r(x)] + \nu[B_r(x)]}}\\
                      &= \lim_{r\to 0^{+}}\frac{\nu[B_r(x)]}{\mu[B_r(x)] + \nu[B_r(x)] - \nu[B_r(x)]} \\
                      &= \lim_{r\to 0^{+}} \frac{\nu[B_r(x)]}{\mu[B_r(x)]}\\
                      &= g
    \end{align*}
    since this holds whever $0 \le f < 1$ it holds $\mu$-almost everywhere and so we conclude.
}

\section{Exercise 2.11}
\emph{
    Let $( \mathbb{R}^{d}, \Sigma, \mu)$ be a finite Borel regular measure space. Let $\mathcal{N}$ be a sub-$\sigma$-algebra
    of $\Sigma$ and let  $\nu := \mu|_{\mathcal{N}}$. Show that if $f \in L^{1}(\mu)$, then there exists a unique $g \in L^{1}(\mathcal{N})$
    such that 
    \[
    \int_{E}^{}f d \mu = \int_{ E }^{} g d \nu
    \]
    for all $E \in \mathcal{N}$
}
\solution{
    First let $f = f^{+} - f^{-}$ with both $f^{\pm} \in L^{1}(\mu)$, then assume $f$ is positive wlog since if not we can construct $g^{+},g^{-}$ and let $g = g^{+} - g^{-}$.
    Define a new measure $\lambda$ on $\mathcal{N}$ as the following, for any set $E \subset \mathcal{N}$
    \[
       \lambda[E] = \int_{E}^{}f d \mu
    \] 
    $\lambda$ is a measure since it is countably additive and sends the null set to 0.
    then $\lambda << \nu = \mu|_{\mathcal{N}}$ since if $\nu[E] = 0$ then the integral is over a null set and is zero so $\lambda[E] = 0$. Thus there exists a function $\tilde f$ measurable with respect to $\mathcal{N}$ such that for all  $E \in \mathcal{N}$
     \[
         \int_{E}^{} f d \mu = \lambda[E] = \int_{E}^{}\tilde f d \nu
    \] 
    thus we have $g = \tilde f$ satisfies our requirement which is unique since uniqueness if guaranteed by the Radon-Nikodym derivative $\nu$-almost everywhere and in $L^{1}(\nu)$ since $f \in L^{1}(\mu)$ so $\lambda( \mathbb{R}^{d}) <\infty$.
}
