\section{Martingale}
Suppose $X_0, X_1,\dots,X_n,\dots$ sequence of random variables on $(\Omega, \mathcal{F})$. \\
Filtration $ \mathcal{F}_0 \subset \mathcal{F}_1 \subset \dots \subset \mathcal{F}_n$ sub $\sigma$-algebra of $ \mathcal{F}$.
\begin{enumerate}
    \item $ \mathbb{E}|X_n| < \infty$ for all $n$.
    \item  $X_n \in \mathcal{F}_n$
    \item $ \mathbb{E}[X_{n+1} | \mathcal{F}_n] =  X_n$
\end{enumerate}
Example
random walk: \\ $\xi_k$ are i.i.da $ \mathbb{E}\xi_k = 0$ and $ \mathbb{E}\xi_k^2 = 1$
\[
    S_n = \sum_{k=1}^{n}\xi_k \qquad \mathcal{F}_n = \sigma(\xi_1, \dots, \xi_n)
\] 
then
\[
S_n \in \mathcal{F}_n
\] 
and
\[
\mathbb{E}|S_n| < \infty
\] 
check that
\[
    \mathbb{E}[S_{n+1}| \mathcal{F}_n] = S_n
\] 
or equivalently check that
\[
    \mathbb{E}[S_{n+1}-S_n| \mathcal{F}_n] = 0
\] 
\[
    = \mathbb{E}[\xi_{n+1} | \mathcal{F}_n] =  \mathbb{E}[\xi_{n+1}] = 0
\] 
Prove that $(S_n^2-n)$ is martingale wrt $ \mathcal{F}_n$. Check
\[
    \mathbb{E}[S_{n+1}^2 - (n+1) - (S_n^2 - n)| \mathcal{F}_n] = \mathbb{E}[\xi_{n+1}^2 + 2 \xi_{n+1} S_n - 1| \mathcal{F}_n]
\] 
\[
    = \mathbb{E}[\xi_{n+1}^2 | \mathcal{F}_n] + \mathbb{E}[2 \xi_{n+1}S_n | \mathcal{F}_n] - 1 = \mathbb{E}[\xi_{n+1}^2] + 2S_n \mathbb{E}[\xi_{n+1}| \mathcal{F}] - 1 = 1 + 0 - 1 = 0
\]
